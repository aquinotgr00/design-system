Dialog (Instant Modal) #

Usage #

Alert #

import dialog from './use-dialog'

dialog.alert({
  title: 'Dialog',
  text : 'Hello This is Dialog',
})

Confirm #

import dialog from './use-dialog'

dialog.confirm({
  title: 'Delete Confirmation',
  text : 'Are you sure?',
}).then((value) => {
  if (value === true) {
    dialog.alert({
      title: 'Info',
      text : 'Deleted',
    })
  }
})
